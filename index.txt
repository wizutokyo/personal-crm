function cst(p){

  if(p.status==="archived")return"archived";
  const r=p.last_contact_date?new Date(p.last_contact_date):new Date(p.created_at||Date.now());

async function callGemini(memo,key){
  const url=`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-lite:generateContent?key=${key}`;
  const res=await fetch(url,{method:"POST",headers:{"Content-Type":"application/json"},body:JSON.stringify({contents:[{parts:[{text:buildPrompt(memo)}]}],generationConfig:{responseMimeType:"application/json"}})});
  if(!res.ok){const e=await res.json().catch(()=>({}));throw new Error(e?.error?.message||`HTTP ${res.status}`)}
  const d=await res.json();
  const t=d?.candidates?.[0]?.content?.parts?.[0]?.text||"";
  return JSON.parse(t.replace(/```json|```/g,"").trim());
}

function ActionsPage({people,onSelect,onToggleNA}){
  const od=people.filter(p=>p.next_action&&!p.next_action_done&&p.next_action_due&&new Date(p.next_action_due)<new Date()).sort((a,b)=>new Date(a.next_action_due)-new Date(b.next_action_due));
  const pend=people.filter(p=>p.next_action&&!p.next_action_done&&(!p.next_action_due||new Date(p.next_action_due)>=new Date())).sort((a,b)=>!a.next_action_due?1:!b.next_action_due?-1:new Date(a.next_action_due)-new Date(b.next_action_due));
  const done=people.filter(p=>p.next_action&&p.next_action_done);

  function Row({p,type}){
    const dov=p.next_action_due?Math.floor((Date.now()-new Date(p.next_action_due).getTime())/86400000):0;
    return(
      <div className="pr" onClick={()=>onSelect(p.id)} style={{borderColor:type==="od"?"rgba(239,68,68,0.44)":"var(--bd)"}}>

        <div style={{textAlign:"right",flexShrink:0}}>
          {p.next_action_due&&<div style={{fontSize:11,color:type==="od"?"var(--er)":"var(--mu)",fontWeight:type==="od"?700:400}}>{p.next_action_due}{type==="od"?` (+${dov}日)`:""}</div>}
          <button className="btng" style={{fontSize:11,padding:"3px 10px",marginTop:4,color:p.next_action_done?"var(--ok)":"var(--mu)"}} onClick={e=>{e.stopPropagation();onToggleNA(p.id)}}>{p.next_action_done?"✓ 完了":"完了にする"}</button>
        </div>
      </div>
    );
  }

  return(
    <div className="pg">

  const ms=months.map(({year,month,label})=>{

    return{label,met:list.length,ref:rc,meet:list.filter(p=>p.has_meeting).length,fri:list.filter(p=>p.is_friend).length};
  });
  const tm=people.length;
  const tr=[...new Set(refs.map(r=>r.person_id))].length;
  const tme=people.filter(p=>p.has_meeting).length;
  const tf=people.filter(p=>p.is_friend).length;
  const tna=people.filter(p=>p.next_action).length;
  const fmt=(n,d)=>d>0?`${(n/d*100).toFixed(1)}%`:"—";
  const od=people.filter(p=>p.next_action&&!p.next_action_done&&p.next_action_due&&new Date(p.next_action_due)<new Date());

  const pm={};people.forEach(p=>{const k=p.met_place||"不明";if(!pm[k])pm[k]={m:0,f:0,me:0};pm[k].m++;if(p.is_friend)pm[k].f++;if(p.has_meeting)pm[k].me++});
  const ps=Object.entries(pm).map(([pl,s])=>({pl,m:s.m,l:s.m>0?(s.f/s.m*100).toFixed(0):0,me:s.m>0?(s.me/s.m*100).toFixed(0):0})).sort((a,b)=>b.m-a.m).slice(0,5);

  const gm={};people.forEach(p=>{const k=p.gender||"other";if(!gm[k])gm[k]={m:0,f:0};gm[k].m++;if(p.is_friend)gm[k].f++});
  const gs=Object.entries(gm).map(([g,s])=>({l:GL[g]||g,m:s.m,land:s.m>0?(s.f/s.m*100).toFixed(0):0}));

  const rm={};refs.forEach(r=>{if(!rm[r.referral_type])rm[r.referral_type]={c:0,f:0};rm[r.referral_type].c++;if(people.find(p=>p.id===r.person_id)?.is_friend)rm[r.referral_type].f++});
  const rs=Object.entries(rm).map(([t,s])=>({t,l:RT[t]?.label||t,e:RT[t]?.emoji||"",col:RT[t]?.color||"var(--ac)",c:s.c,land:s.c>0?(s.f/s.c*100).toFixed(0):0}));

  const mx=Math.max(...ms.map(m=>m.met),1);
  const bc=["var(--ac)","#34d399","#fbbf24","#f472b6"];
  const bk=["met","ref","meet","fri"];
  const bl=["出会い","送客","面談","友達登録"];

  function MBar({label,pct,color,sub}){
    return(
      <div style={{marginBottom:10}}>
        <div style={{display:"flex",justifyContent:"space-between",fontSize:12,marginBottom:3}}>
          <span style={{fontWeight:600}}>{label}</span>
          <span style={{color:color||"var(--ac)",fontWeight:700}}>{pct}%</span>
        </div>
        <div style={{height:5,background:"var(--sf3)",borderRadius:3,overflow:"hidden"}}>
          <div style={{height:"100%",width:`${Math.min(Number(pct),100)}%`,background:color||"var(--ac)",borderRadius:3}}/>
        </div>
        {sub&&<div style={{fontSize:10,color:"var(--dim)",marginTop:2}}>{sub}</div>}
      </div>
    );
  }

  return(
    <div className="pg">
      <div style={{display:"flex",justifyContent:"space-between",alignItems:"flex-start",marginBottom:20,flexWrap:"wrap",gap:8}}>
        <div>
          <h2 style={{fontFamily:"'Syne',sans-serif",fontSize:22,fontWeight:800}}>ダッシュボード</h2>
          <p style={{color:"var(--mu)",fontSize:13}}>{now.getFullYear()}年{now.getMonth()+1}月{now.getDate()}日 現在</p>
        </div>
        <button className="btn" onClick={()=>setPage("input")}>＋ 新規入力</button>
      </div>

      <div className="fw" style={{marginBottom:8}}>
        {[["総出会い数",tm,"var(--tx)","人"],["送客人数",tr,"#60a5fa","人"],["面談獲得",tme,"#fbbf24","人"],["友達登録",tf,"#22c55e","人"],["NA設定",tna,"#a78bfa","人"]].map(([l,v,c,s])=>(
          <div key={l} className="kc"><div className="kl">{l}</div><div className="kv" style={{color:c}}>{v}</div><div className="ks">{s}</div></div>
        ))}
      </div>
      <div className="fw" style={{marginBottom:16}}>
        {[["送客率",fmt(tr,tm),"#60a5fa","送客÷出会い"],["着地率",fmt(tf,tr),"#34d399","友達÷送客"],["面談率",fmt(tme,tr),"#fbbf24","面談÷送客"],["出会い面談率",fmt(tme,tm),"#f472b6","面談÷出会い"],["NA率",fmt(tna,tm),"#a78bfa","NA設定÷出会い"]].map(([l,v,c,s])=>(
          <div key={l} className="kc"><div className="kl">{l}</div><div className="kv" style={{color:c}}>{v}</div><div className="ks">{s}</div></div>
        ))}
      </div>

      <div className="g2" style={{marginBottom:10}}>
        <div className="card">
          <div style={{fontSize:13,fontWeight:700,marginBottom:10}}>月別推移（直近6ヶ月）</div>
          <div className="fw" style={{marginBottom:8,gap:6}}>
            {bk.map((k,i)=><div key={k} style={{display:"flex",alignItems:"center",gap:3,fontSize:10,color:"var(--mu)"}}><span style={{width:8,height:8,borderRadius:2,background:bc[i],display:"inline-block"}}/>{bl[i]}</div>)}
          </div>
          <div style={{display:"flex",alignItems:"flex-end",gap:6,height:90}}>
            {ms.map((m,mi)=>(
              <div key={mi} style={{flex:1,display:"flex",flexDirection:"column",alignItems:"center"}}>
                <div style={{width:"100%",display:"flex",gap:1,alignItems:"flex-end",height:78}}>
                  {bk.map((k,i)=><div key={k} style={{flex:1,height:`${mx>0?(m[k]/mx*78):0}px`,background:bc[i],borderRadius:"2px 2px 0 0",minHeight:m[k]>0?"3px":"0",opacity:.85}} title={`${bl[i]}: ${m[k]}`}/>)}
                </div>
                <div style={{fontSize:10,color:"var(--dim)",marginTop:3}}>{m.label}</div>
              </div>
            ))}
          </div>
        </div>

        <div className="card">
          <div style={{fontSize:13,fontWeight:700,marginBottom:12}}>場所別 着地率</div>
          {ps.length===0?<div style={{color:"var(--dim)",fontSize:13}}>データなし</div>:ps.map(s=><MBar key={s.pl} label={`${s.pl} (${s.m}人)`} pct={s.l} color="#34d399" sub={`面談率: ${s.me}%`}/>)}
        </div>

        <div className="card">
          <div style={{fontSize:13,fontWeight:700,marginBottom:12}}>性別別 着地率</div>
          {gs.length===0?<div style={{color:"var(--dim)",fontSize:13}}>データなし</div>:gs.map(s=><MBar key={s.l} label={`${s.l} (${s.m}人)`} pct={s.land} color="#f472b6"/>)}
        </div>

        <div className="card">
          <div style={{fontSize:13,fontWeight:700,marginBottom:12}}>送客先別 着地率</div>
          {rs.length===0?<div style={{color:"var(--dim)",fontSize:13}}>データなし</div>:rs.map(s=><MBar key={s.t} label={`${s.e} ${s.l} (${s.c}件)`} pct={s.land} color={s.col}/>)}
        </div>
      </div>

      {od.length>0&&(
        <div className="card" style={{borderColor:"rgba(239,68,68,0.33)"}}>
          <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:10}}>
            <div style={{fontSize:13,fontWeight:700,color:"var(--er)"}}>⚠ 期限切れアクション ({od.length}件)</div>
            <button className="btng" style={{fontSize:11,padding:"4px 12px"}} onClick={()=>setPage("actions")}>全て見る →</button>
          </div>
          {od.slice(0,4).map(p=>(
            <div key={p.id} style={{display:"flex",gap:12,fontSize:12,padding:"6px 0",borderBottom:"1px solid var(--bd)"}}>
              <span style={{fontWeight:700,minWidth:60}}>{p.name}</span>
              <span style={{color:"var(--mu)",flex:1,overflow:"hidden",textOverflow:"ellipsis",whiteSpace:"nowrap"}}>{p.next_action}</span>
              <span style={{color:"var(--er)",fontWeight:700,flexShrink:0}}>{p.next_action_due}</span>
            </div>
          ))}
        </div>
      )}

      {people.length===0&&(
        <div className="card" style={{textAlign:"center",padding:60,marginTop:16}}>
          <div style={{fontSize:36,marginBottom:12}}>✦</div>
          <div style={{fontWeight:700,fontSize:16,marginBottom:6}}>まだデータがありません</div>
          <div style={{color:"var(--mu)",fontSize:13,marginBottom:20}}>最初の出会いを記録してみましょう</div>
          <button className="btn" onClick={()=>setPage("input")}>＋ 最初の人物を入力する</button>
        </div>
      )}
    </div>
  );
}

function App(){
  const [page,setPage]=useState("dashboard");
  const [sub,setSub]=useState(null);
  const [people,setPeople]=useState(()=>DB.people());
  const [refs,setRefs]=useState(()=>DB.refs());
  const [parsed,setParsed]=useState(null);
  const [origMemo,setOrigMemo]=useState("");
  const [toastState,setToastState]=useState({msg:"",type:"success"});

  useEffect(()=>{DB.sp(people)},[people]);
  useEffect(()=>{DB.sr(refs)},[refs]);

  function showToast(msg,type="success"){
    setToastState({msg,type});
    setTimeout(()=>setToastState({msg:"",type:"success"}),3000);
  }

  const odCount=people.filter(p=>p.next_action&&!p.next_action_done&&p.next_action_due&&new Date(p.next_action_due)<new Date()).length;

  function navTo(p){setSub(null);setPage(p);window.scrollTo(0,0)}

  function handleAIParsed(data,memo){setParsed(data);setOrigMemo(memo);setSub("confirm")}

  function handleSaveNew(f){
    const id=uid();const now=new Date().toISOString();
    setPeople(prev=>[{
      id,name:f.name,met_date:f.met_date||null,met_place:f.met_place||null,
      age:f.age?Number(f.age):null,gender:f.gender||null,occupation:f.occupation||null,
      residence:f.residence||null,university:f.university||null,hometown:f.hometown||null,
      family_structure:f.family_structure||null,conversation:f.conversation||null,
      next_action:f.next_action||null,next_action_due:f.next_action_due||null,next_action_done:false,
      has_meeting:f.has_meeting||false,meeting_date:f.meeting_date||null,
      is_friend:f.is_friend||false,friended_date:f.friended_date||null,
      status:"active",notes:f.notes||null,
      last_contact_date:f.met_date||now.split("T")[0],created_at:now,updated_at:now,
    },...prev]);
    if(f.referral_types?.length>0){
      setRefs(prev=>[...prev,...f.referral_types.map(type=>({id:uid(),person_id:id,referral_type:type,referral_date:f.met_date||now.split("T")[0],created_at:now}))]);
    }
    showToast(`${f.name}を登録しました`);
    setSub(null);setPage("people");
  }

  function handleUpdate(id,f){
    setPeople(prev=>prev.map(p=>p.id!==id?p:{
      ...p,name:f.name,met_date:f.met_date||null,met_place:f.met_place||null,
      age:f.age?Number(f.age):null,gender:f.gender||null,occupation:f.occupation||null,
      residence:f.residence||null,university:f.university||null,hometown:f.hometown||null,
      family_structure:f.family_structure||null,conversation:f.conversation||null,
      next_action:f.next_action||null,next_action_due:f.next_action_due||null,next_action_done:f.next_action_done||false,
      has_meeting:f.has_meeting||false,meeting_date:f.meeting_date||null,
      is_friend:f.is_friend||false,friended_date:f.friended_date||null,
      status:f.status||"active",notes:f.notes||null,updated_at:new Date().toISOString(),
    }));
    setRefs(prev=>[...prev.filter(r=>r.person_id!==id),...(f.referral_types||[]).map(type=>({id:uid(),person_id:id,referral_type:type,referral_date:f.met_date||new Date().toISOString().split("T")[0],created_at:new Date().toISOString()}))]);
    showToast("保存しました");setSub(`detail:${id}`);
  }

  function handleDelete(id){
    const p=people.find(x=>x.id===id);
    setPeople(prev=>prev.filter(x=>x.id!==id));
    setRefs(prev=>prev.filter(r=>r.person_id!==id));
    showToast(`${p?.name}を削除しました`);
    setSub(null);setPage("people");
  }

  function handleToggleNA(id){setPeople(prev=>prev.map(p=>p.id===id?{...p,next_action_done:!p.next_action_done}:p))}

  const navPage=sub?.startsWith("detail")||sub?.startsWith("edit")?"people":page;

  function renderContent(){
    if(sub==="confirm")return <ConfirmPage parsed={parsed} memo={origMemo} onSave={handleSaveNew} onBack={()=>setSub(null)} toast={showToast}/>;
    if(sub?.startsWith("detail:")){
      const id=sub.split(":")[1];
      return <PersonDetail pid={id} people={people} refs={refs} onBack={()=>setSub(null)} onEdit={()=>setSub(`edit:${id}`)} onDelete={handleDelete} onToggleNA={handleToggleNA}/>;
    }
    if(sub?.startsWith("edit:")){
      const id=sub.split(":")[1];
      return <EditPage pid={id} people={people} refs={refs} onSave={handleUpdate} onBack={()=>setSub(`detail:${id}`)} toast={showToast}/>;
    }
    if(page==="dashboard")return <Dashboard people={people} refs={refs} setPage={navTo}/>;
    if(page==="input")return <InputPage onConfirm={handleAIParsed} toast={showToast}/>;
    if(page==="people")return <PeoplePage people={people} refs={refs} onSelect={id=>setSub(`detail:${id}`)} onNew={()=>navTo("input")}/>;
    if(page==="actions")return <ActionsPage people={people} onSelect={id=>setSub(`detail:${id}`)} onToggleNA={handleToggleNA}/>;
    return null;
  }

  return(
    <>
      <Nav page={navPage} setPage={navTo} od={odCount}/>
      <main>{renderContent()}</main>
      <Toast msg={toastState.msg} type={toastState.type}/>
    </>
  );
}

ReactDOM.createRoot(document.getElementById("root")).render(<App/>);
</script>
</body>
</html>
